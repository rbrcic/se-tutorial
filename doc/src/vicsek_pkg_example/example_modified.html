21def update_model():
22    global pos, theta
23    
24    # 1. Calculate all-to-all distances using broadcasting
25    # dx.shape will be (n, n)
26    dx = pos[:, 0, np.newaxis] - pos[:, 0]
27    dy = pos[:, 1, np.newaxis] - pos[:, 1]
28    
29    # Apply periodic distance (closest path in a wrap-around box)
30    dx = (dx + 0.5) % 1 - 0.5
31    dy = (dy + 0.5) % 1 - 0.5
32    
33    dist = np.sqrt(dx**2 + dy**2)
34    
35    # 2. Find neighbors within interaction radius
36    adj = dist < r_interact
37    
38    # 3. Average the angles of neighbors
39    new_theta = theta.copy()
40    for i in range(n):
41        neighbors_theta = theta[adj[i]]
42        avg_sin = np.mean(np.sin(neighbors_theta))
43        avg_cos = np.mean(np.cos(neighbors_theta))
44        new_theta[i] = np.arctan2(avg_sin, avg_cos)
45    
46    # 4. Add noise and update heading
47    theta = new_theta + eta * np.random.uniform(-np.pi, np.pi, n)
48    
49    # 5. Move particles
50    pos[:, 0] += v * np.cos(theta) * dt
51    pos[:, 1] += v * np.sin(theta) * dt
52    
53    # 6. Periodic Boundary Conditions
54    pos = pos % 1
55
