 51def update_model():
 52    global r, theta, counter
 53    if running:
 54        for i in range(vicsek.n):
 55            sum_sin = 0
 56            sum_cos = 0
 57            neighbours = 0
 58
 59            for j in range(vicsek.n):
 60                if i != j:
 61                    if distance(r[i], r[j]) < vicsek.d:
 62                        theta_j = 2 * np.pi * theta[j]
 63                        sum_sin = sum_sin + np.sin(theta_j)
 64                        sum_cos = sum_cos + np.cos(theta_j)
 65                        neighbours = neighbours + 1
 66
 67            if neighbours > 0:
 68                avg_theta = np.arctan2(sum_sin / neighbours, sum_cos / neighbours)
 69                theta[i] = (avg_theta / (2 * np.pi)) + vicsek.eta * (np.random.rand() - 0.5)
 70
 71            dx = vicsek.v * vicsek.dt * np.cos(2 * np.pi * theta[i])
 72            dy = vicsek.v * vicsek.dt * np.sin(2 * np.pi * theta[i])
 73
 74            r[i, 0] = r[i, 0] + dx
 75            r[i, 1] = r[i, 1] + dy
 76
 77            if r[i, 0] > 1:
 78                r[i, 0] = 0
 79            if r[i, 1] > 1:
 80                r[i, 1] = 0
 81            if r[i, 0] < 0:
 82                r[i, 0] = 1
 83            if r[i, 1] < 0:
