 64def update_model():
 65    global r, theta, counter
 66    if running:
 67        for i in range(vicsek.n):
 68            sum_sin = 0
 69            sum_cos = 0
 70            neighbours = 0
 71
 72            for j in range(vicsek.n):
 73                if i != j:
 74                    if distance(r[i], r[j]) < vicsek.d:
 75                        theta_j = 2 * np.pi * theta[j]
 76                        sum_sin = sum_sin + np.sin(theta_j)
 77                        sum_cos = sum_cos + np.cos(theta_j)
 78                        neighbours = neighbours + 1
 79
 80            if neighbours > 0:
 81                avg_theta = np.arctan2(sum_sin / neighbours, sum_cos / neighbours)
 82                theta[i] = (avg_theta / (2 * np.pi)) + vicsek.eta * (np.random.rand() - 0.5)
 83
 84            dx = vicsek.v * vicsek.dt * np.cos(2 * np.pi * theta[i])
 85            dy = vicsek.v * vicsek.dt * np.sin(2 * np.pi * theta[i])
 86
 87            r[i, 0] = r[i, 0] + dx
 88            r[i, 1] = r[i, 1] + dy
